# checks for :

# not require a docstring

# module-rgx=(([a-z_][a-z0-9_]*)|([a-zA-Z0-9]+))$

# class-rgx=[A-Z_][a-zA-Z0-9]+$

# function-rgx=[a-z_][a-z0-9_]{2,30}$

#method-rgx=[a-z_][a-z0-9_]{2,30}$
